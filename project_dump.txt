from __future__ import annotations

from pathlib import Path
from urllib.parse import urlparse, unquote

import requests


def download_generic(url: str, timeout_sec: int = 120) -> bytes:
    """
    Generic downloader:
    - http/https: requests GET
    - file:///: local file read (bytes)
    """
    u = (url or "").strip()

    # Local file support: file:///C:/... vagy file:/...
    if u.lower().startswith("file:"):
        parsed = urlparse(u)
        # Windows: parsed.path = '/C:/Tarhely/...'
        path_str = unquote(parsed.path)

        # vedd le az elejĂ©rĹ‘l a vezetĹ‘ perjelet, ha Windows drive-os
        if len(path_str) >= 3 and path_str[0] == "/" and path_str[2] == ":":
            path_str = path_str[1:]

        p = Path(path_str)
        if not p.exists():
            raise FileNotFoundError(f"File URL not found: {u} -> {p}")

        return p.read_bytes()

    # Default: HTTP(S)
    r = requests.get(u, headers={"User-Agent": "shop-sync/1.0"}, timeout=timeout_sec)
    r.raise_for_status()
    return r.content

from __future__ import annotations
from typing import Callable, Dict, Optional

DownloaderFn = Callable[[str, int], bytes]  # (url, timeout_sec) -> bytes

_DOWNLOADERS: Dict[str, DownloaderFn] = {}


class UnknownDownloaderError(ValueError):
    pass


def register_downloader(name: str):
    if not name or not name.strip():
        raise ValueError("downloader name cannot be empty")
    key = name.strip().lower()

    def _decorator(fn: DownloaderFn) -> DownloaderFn:
        _DOWNLOADERS[key] = fn
        return fn

    return _decorator


def get_downloader(supplier_name: str) -> Optional[DownloaderFn]:
    return _DOWNLOADERS.get((supplier_name or "").strip().lower())


def list_downloaders() -> list[str]:
    return sorted(_DOWNLOADERS.keys())

import csv
from io import StringIO
from typing import Any, Dict, List


def parse_csv_bytes(
    data: bytes,
    *,
    encoding: str = "utf-8",
    delimiter: str = ",",
) -> List[Dict[str, Any]]:
    """
    CSV fĂˇjl feldolgozĂˇsa byte formĂˇtumbĂłl.

    A fĂĽggvĂ©ny egy byte formĂˇtumban kapott CSV ĂˇllomĂˇnyt alakĂ­t Ăˇt
    Python szĂłtĂˇrak listĂˇjĂˇvĂˇ. A fejlĂ©c sor alapjĂˇn kulcs-Ă©rtĂ©k
    pĂˇrokat hoz lĂ©tre minden egyes adatsorhoz.

    Tipikus felhasznĂˇlĂˇs:
        - FastAPI UploadFile.read()
        - API-n keresztĂĽl kapott CSV tartalom
        - hĂˇttĂ©rfolyamatban tĂ¶rtĂ©nĹ‘ CSV import

    ParamĂ©terek:
        data (bytes):
            A CSV fĂˇjl teljes tartalma byte formĂˇtumban.

        encoding (str, alapĂ©rtelmezett: "utf-8"):
            A fĂˇjl karakterkĂłdolĂˇsa. HibĂˇs karakterek esetĂ©n
            a rendszer nem dob kivĂ©telt, hanem lecserĂ©li azokat.

        delimiter (str, alapĂ©rtelmezett: ","):
            A CSV mezĹ‘elvĂˇlasztĂł karaktere (pl. ",", ";", "\t").

    VisszatĂ©rĂ©si Ă©rtĂ©k:
        List[Dict[str, Any]]:
            Egy lista, amelyben minden elem egy CSV sor.
            A kulcsok a fejlĂ©c oszlopnevei,
            az Ă©rtĂ©kek az adott sor cellaĂ©rtĂ©kei (stringkĂ©nt).

    MegjegyzĂ©sek:
        - Minden mezĹ‘ string tĂ­pusĂş lesz.
        - TĂ­puskonverziĂłt kĂĽlĂ¶n kell vĂ©gezni (pl. int(), float()).
        - A CSV-nek tartalmaznia kell fejlĂ©c sort.
        - Ăśres mezĹ‘k ĂĽres stringkĂ©nt ("") jelennek meg.
    """

    # Byte adat dekĂłdolĂˇsa szĂ¶veggĂ© a megadott encoding szerint.
    # HibĂˇs karakterek esetĂ©n nem dob hibĂˇt, hanem lecserĂ©li azokat.
    text = data.decode(encoding, errors="replace")

    # A szĂ¶veget egy memĂłriabeli "fĂˇjllĂˇ" alakĂ­tjuk,
    # hogy a csv.DictReader fĂˇjlkĂ©nt tudja kezelni.
    f = StringIO(text)

    # DictReader hasznĂˇlata:
    # - Az elsĹ‘ sort fejlĂ©cnek tekinti
    # - Minden tovĂˇbbi sort dict formĂˇban ad vissza
    reader = csv.DictReader(f, delimiter=delimiter)

    # A reader iterĂˇlhatĂł objektum, ezĂ©rt listĂˇba gyĹ±jtjĂĽk.
    # Minden sorbĂłl explicit dict-et kĂ©szĂ­tĂĽnk.
    return [dict(row) for row in reader]

import csv
import io


def parse_csv_bytes(
    data: bytes,
    *,
    encoding: str = "utf-8",
    delimiter: str = ";",
    has_header: bool = True,
) -> List[Dict[str, Any]]:
    """
    CSV bytes -> List[Dict[str, Any]]

    - kezeli a UTF-8 BOM-ot
    - delimiter paramĂ©terezhetĹ‘
    - ha has_header=False, akkor oszlopnevek: col1, col2, ...
    """
    # BOM-barĂˇt decode
    text = data.decode(encoding, errors="replace")
    if text.startswith("\ufeff"):
        text = text.lstrip("\ufeff")

    f = io.StringIO(text)

    reader = csv.reader(f, delimiter=delimiter)

    rows: List[Dict[str, Any]] = []

    try:
        first = next(reader)
    except StopIteration:
        return []

    if has_header:
        headers = [str(h).strip() for h in first]
    else:
        headers = [f"col{i+1}" for i in range(len(first))]
        rows.append({headers[i]: first[i] for i in range(len(headers))})

    for r in reader:
        # rĂ¶videbb sor esetĂ©n pad-eljĂĽk, hosszabbnĂˇl vĂˇgjuk
        r2 = list(r[: len(headers)]) + [""] * max(0, len(headers) - len(r))
        rows.append({headers[i]: r2[i] for i in range(len(headers))})

    return rows

def cache_write(supplier_name: str, content: bytes) -> Path:
    """
    LetĂ¶ltĂ¶tt CSV tartalom mentĂ©se cache mappĂˇba.
    """
    CACHE_DIR.mkdir(parents=True, exist_ok=True)

    ts = time.strftime("%Y%m%d_%H%M%S")
    path = CACHE_DIR / supplier_name / f"{ts}.csv"
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_bytes(content)
    return path


def _download_supplier_csv(*, supplier_name: str, url: str, timeout_sec: int = 120) -> bytes:
    """
    Registry-s letĂ¶ltĂ©s:
    - ha van supplier-specifikus downloader -> azt hasznĂˇlja
    - kĂĽlĂ¶nben generic downloader
    """
    fn = get_downloader(supplier_name)
    if fn is None:
        return download_generic(url, timeout_sec=timeout_sec)
    return fn(url, timeout_sec)


def _get_latest_cache_file(supplier_name: str) -> Optional[Path]:
    supplier_cache_dir = CACHE_DIR / supplier_name
    if not supplier_cache_dir.exists():
        return None

    files = sorted(
        supplier_cache_dir.glob("*.csv"),
        key=lambda p: p.stat().st_mtime,
        reverse=True,
    )

    return files[0] if files else None


def _is_cache_fresh(path: Path, ttl_seconds: int) -> bool:
    age = time.time() - path.stat().st_mtime
    return age <= ttl_seconds


def ingest_one_supplier_csv(supplier_name: str) -> List[Dict[str, Any]]:
    supplier_dir = SUPPLIERS_DIR / supplier_name
    if not supplier_dir.exists():
        raise FileNotFoundError(f"Nincs ilyen mappa: {supplier_dir}")

    cfg = load_supplier_csv_config(supplier_dir)

    test_mode = (os.getenv("TEST_MODE") or "").strip() == "1"

    data: bytes

    # ------------------------------------------------
    # TEST MODE: cache-bĹ‘l dolgozzunk, ha friss
    # ------------------------------------------------
    if test_mode and cfg.cache.enabled:
        latest_cache = _get_latest_cache_file(cfg.name)

        if latest_cache and _is_cache_fresh(latest_cache, cfg.cache.ttl_seconds):
            print(f"[{cfg.name}] TEST MODE cache HIT â†’ {latest_cache.name}")
            data = latest_cache.read_bytes()
        else:
            print(f"[{cfg.name}] TEST MODE cache MISS â†’ downloading")
            data = _download_supplier_csv(
                supplier_name=cfg.name,
                url=cfg.url,
                timeout_sec=120,
            )
            cache_write(cfg.name, data)

    # ------------------------------------------------
    # NORMAL MODE (marad a jelenlegi logika)
    # ------------------------------------------------
    else:
        data = _download_supplier_csv(
            supplier_name=cfg.name,
            url=cfg.url,
            timeout_sec=120,
        )

        if cfg.cache.enabled:
            cache_write(cfg.name, data)

    # ------------------------------------------------
    # CSV parse
    # ------------------------------------------------
    rows = parse_csv_bytes(
        data,
        encoding=cfg.encoding,
        delimiter=cfg.delimiter,
        has_header=cfg.has_header,
    )

    for r in rows:
        r["_supplier"] = cfg.name

    return rows

from __future__ import annotations

from typing import Any, Dict, Optional, Sequence

Product = Dict[str, Any]


def _copy_if_empty(dst: Product, src: Product, dst_key: str, src_key: str) -> None:
    dv = dst.get(dst_key)
    if dv is None or (isinstance(dv, str) and not dv.strip()):
        sv = src.get(src_key)
        if sv is not None:
            dst[dst_key] = sv


def _ensure_image_urls(dst: Product, src: Product) -> None:
    # HaldepĂł: image_url or image_urls -> master.image_urls
    urls = None
    if isinstance(src.get("image_urls"), list) and src["image_urls"]:
        urls = [str(x).strip() for x in src["image_urls"] if str(x).strip()]
    else:
        u = str(src.get("image_url") or src.get("image") or "").strip()
        if u:
            urls = [u]

    if urls:
        dst["image_urls"] = urls


class HaldepoEnrichPlugin(EnrichPlugin):
    name = "haldepo"

    def build_indexes(self, supplier_rows: Sequence[Product], *, master_keys: Sequence[str]) -> Any:
        return build_default_index(supplier_rows, master_keys=master_keys)

    def find_supplier_row(self, index: Any, master: Product) -> Optional[Product]:
        # index_enricher_by_key usually returns a callable or dict-like structure depending on your implementation.
        # In your dump it's used like: hit = index(master)
        try:
            return index(master)  # type: ignore[misc]
        except TypeError:
            # if index is dict-like and you have a match_key in master:
            mk = (master.get("match_key") or "").strip()
            if mk:
                return index.get(mk)  # type: ignore[union-attr]
            return None

    def apply(self, master: Product, supplier_row: Product) -> Product:
        p = dict(master)

        # --- text fields
        _copy_if_empty(p, supplier_row, "name_hu", "name_hu")
        _copy_if_empty(p, supplier_row, "name_hu", "name")
        _copy_if_empty(p, supplier_row, "description_hu", "description_hu")
        _copy_if_empty(p, supplier_row, "description_hu", "description")

        # --- gtin/ean
        _copy_if_empty(p, supplier_row, "gtin", "gtin")
        _copy_if_empty(p, supplier_row, "gtin", "ean")

        # --- images: HaldepĂł legyen erĹ‘s forrĂˇs
        _ensure_image_urls(p, supplier_row)

        return p

class HaldepoWholesalePlugin:

        for r in rows:

        # Natura alap

        wp = master.get("wholesale_price")

        h = indexes.get("by_model", {}).get(model)
        if not h:
            return WholesaleResult(
                gross_price=float(gp) if gp is not None else None,
                wholesale_price=float(wp) if wp is not None else None,
                supplier="natura",
            )

        # HaldepĂł felĂĽlĂ­r
        if h.get("gross_price") is not None:
            gp = h["gross_price"]
        if h.get("wholesale_price") is not None:

        # HaldepĂł wholesale nĂˇlad nettĂł -> bruttĂł + 5-re felfelĂ©

        return WholesaleResult(

        )

from __future__ import annotations

from dataclasses import dataclass
from typing import Any, Dict, List, Mapping, Optional, Sequence, Tuple

# -------------------------------------------------
# Minimal helpers (self-contained)

def build_master_keys(master_products: Sequence[Product]) -> List[str]:
    """
    MegnĂ©zi a master mintĂˇn, hogy mely kulcsok alapjĂˇn szoktunk tudni pĂˇrosĂ­tani.
    NĂˇlad jellemzĹ‘en: match_key / model / gtin/ean.
    """
    # preferĂˇlt sorrend
    candidates = ["match_key", "model", "modelNumber", "gtin", "ean", "sku"]
    present: List[str] = []
    for k in candidates:
        for p in master_products[:200]:  # elĂ©g mintĂˇt nĂ©zni
            v = p.get(k)
            if v is None:
                continue
            if isinstance(v, str) and v.strip():
                present.append(k)
                break
            if not isinstance(v, str):
                present.append(k)
                break
    # duplikĂˇciĂł kiszĹ±rĂ©s, sorrend megtartĂˇs
    seen = set()
    out = []
    for k in present:
        if k not in seen:
            seen.add(k)
            out.append(k)
    return out


def _normalize_str(x: Any) -> str:
    return str(x or "").strip().lower()


def index_enricher_by_key(rows: Sequence[Product], master_keys: Sequence[str]) -> Any:
    """
    EgyszerĹ± index: minden supplier row-hoz kĂ©pezĂĽnk kulcsot a master_keys szerint.
    Visszaad egy callable-t: index(master_product) -> supplier_row|None
    """
    lookup: Dict[Tuple[str, str], Product] = {}

    for r in rows:
        for k in master_keys:
            v = _normalize_str(r.get(k))
            if v:
                lookup[(k, v)] = r

    def find(master: Product) -> Optional[Product]:
        for k in master_keys:
            v = _normalize_str(master.get(k))
            if not v:
                continue
            hit = lookup.get((k, v))
            if hit:
                return hit
        return None

    return find

import re

_WS = re.compile(r"[\s\-_\/]+", re.UNICODE)

def normalize_match_key(value: str | None) -> str:
    if not value:
        return ""
    s = value.strip().upper()
    s = _WS.sub("", s)
    s = re.sub(r"[^A-Z0-9]", "", s)
    return s

        raise NotImplementedError


def enrich_products(
    *,

def normalize_rows(
    supplier_name: str,
    raw_rows: List[Dict[str, Any]],
    *,
    ensure_supplier_meta: bool = True,
) -> List[Dict[str, Any]]:

    Ez az a fĂĽggvĂ©ny, amit a rendszer tĂ¶bbi rĂ©sze hĂ­v (export/merge elĹ‘tt),
    Ă©s nem kell tudnia arrĂłl, hogy konkrĂ©tan melyik supplier modul mit csinĂˇl.

    ParamĂ©terek:
        supplier_name (str):
            A beszĂˇllĂ­tĂł azonosĂ­tĂł neve (pl. "natura").
            Ennek egyeznie kell a regisztrĂˇlt normalizĂˇlĂł nevĂ©vel.

        raw_rows (List[Dict[str, Any]]):
            Nyers sorok (CSV parse utĂˇn).

        ensure_supplier_meta (bool):
            Ha True, akkor ellenĹ‘rzi/rĂˇsegĂ­t, hogy minden normalizĂˇlt sorban
            legyen '_supplier' mezĹ‘. (Merge Ă©s debug miatt hasznos.)
            AlapĂ©rtelmezett: True

    VisszatĂ©rĂ©si Ă©rtĂ©k:
        List[Dict[str, Any]]:
            NormalizĂˇlt sorok listĂˇja (supplier-specifikus mapping alapjĂˇn).

    KivĂ©tel:
        UnknownSupplierError:
            Ha nincs regisztrĂˇlt normalizĂˇlĂł.

    MegjegyzĂ©s:
        A registry mĹ±kĂ¶dĂ©sĂ©hez a supplier modulokat be kell importĂˇlni.

                import src.normalize.suppliers

    fn = get_normalizer(supplier_name)

    # Lefuttatjuk a beszĂˇllĂ­tĂł-specifikus normalizĂˇlĂłt

    return normalized

    created = updated = failed = 0
    i = 0

    # JSONL log (minden sor egy JSON objektum)
    with open(log_path, "a", encoding="utf-8") as log:
        for p in products:
            i += 1
            if max_items is not None and i > max_items:
                break

            sku = str(p.get("sku", "")).strip()

            try:
                # Payload Ă©pĂ­tĂ©s beszĂˇllĂ­tĂłi termĂ©kbĹ‘l
                payload = build_payload(p)
                action = None

                # DĂ¶ntĂ©s: update vagy create
                existing_id = sku_map.get(payload["sku"])
                if existing_id:
                    # UPDATE
                    client.update_product(existing_id, payload)
                    action = "updated"
                    updated += 1
                else:
                    # CREATE
                    resp = client.create_product(payload)
                    sku_map[payload["sku"]] = resp.get("id")
                    action = "created"
                    created += 1

                print(f"{i} {sku} -> {action}")
                log.write(json.dumps({"sku": sku, "action": action}, ensure_ascii=False) + "\n")

            except HTTPError as e:
                # HTTP szintĹ± hiba (4xx/5xx)
                failed += 1
                status = getattr(e.response, "status_code", None)
                body = getattr(e.response, "text", "")[:2000]

                print(f"{i} {sku} -> FAILED ({status})")

                # HibĂˇs payload mentĂ©se kĂ©sĹ‘bbi elemzĂ©shez
                try:
                    os.makedirs("data/debug/failed_payloads", exist_ok=True)
                    with open(f"data/debug/failed_payloads/{sku}.json", "w", encoding="utf-8") as f:
                        json.dump(payload, f, ensure_ascii=False, indent=2)
                except Exception:
                    # MentĂ©si hiba ne ĂˇllĂ­tsa meg a futĂˇst
                    pass

                # JSONL log sor
                log.write(
                    json.dumps(
                        {"sku": sku, "action": "failed", "status": status, "body": body},
                        ensure_ascii=False,
                    )
                    + "\n"
                )

                # Rate limit esetĂ©n agresszĂ­vebb backoff
                if status == 429:
                    time.sleep(max(2.0, sleep_s * 5))
                else:
                    time.sleep(sleep_s)

            except Exception as e:
                # EgyĂ©b futĂˇsidejĹ± hiba
                failed += 1
                print(f"{i} {sku} -> FAILED (exception): {e}")
                log.write(json.dumps({"sku": sku, "action": "failed", "error": str(e)}, ensure_ascii=False) + "\n")
                time.sleep(sleep_s)

            # ĂltalĂˇnos throttling kĂ©t termĂ©k kĂ¶zĂ¶tt
            time.sleep(sleep_s)

    return {"created": created, "updated": updated, "failed": failed, "total": i}


def chunked(lst: List[Any], n: int) -> Iterable[List[Any]]:
    """
    Lista feldarabolĂˇsa n mĂ©retĹ± darabokra (chunkokra).

    ParamĂ©terek:
        lst (List[Any]): FeldarabolandĂł lista.
        n (int): Chunk mĂ©ret.

    Yields:
        List[Any]: A kĂ¶vetkezĹ‘ chunk (max n elem).
    """
    for i in range(0, len(lst), n):
        yield lst[i : i + n]


def post_batch(session: requests.Session, base_url: str, reqs: List[Dict[str, Any]]) -> Dict[str, Any]:
    """
    Shoprenter /batch endpoint meghĂ­vĂˇsa.

    A Shoprenter batch formĂˇtum:
        {

def build_batch_requests_for_products(
    *,
    base_url: str,
    sku_map: Dict[str, str],
    payloads: List[Dict[str, Any]],
) -> List[Dict[str, Any]]:
    """
    Batch request lista Ă©pĂ­tĂ©se termĂ©k payloadokbĂłl (UPsert logika).

    SzabĂˇly:
        - ha SKU benne van a sku_map-ben -> PUT /productExtend/{id}
        - ha nincs -> POST /productExtend

    Fontos:
        Itt a PUT/POST requestek "data" mezĹ‘jĂ©be jelenleg a *teljes payload* kerĂĽl.
        Sok esetben Ă©rdemes PUT esetĂ©n csak update mezĹ‘ket kĂĽldeni
        (pl. build_update_payload_from_full), hogy elkerĂĽld a nem kĂ­vĂˇnt felĂĽlĂ­rĂˇsokat.

    ParamĂ©terek:
        base_url (str): Shoprenter API base URL.
        sku_map (Dict[str, str]): SKU -> productExtend id map.
        payloads (List[Dict[str, Any]]): TermĂ©k payloadok (mind tartalmazza a "sku"-t).

    VisszatĂ©rĂ©si Ă©rtĂ©k:
        List[Dict[str, Any]]: /batch-be kĂĽldhetĹ‘ request elemek listĂˇja.
    """
    reqs: List[Dict[str, Any]] = []

    base = base_url.rstrip("/")

    for payload in payloads:
        sku = payload["sku"].strip()
        existing_id = sku_map.get(sku)

        if existing_id:
            reqs.append(
                {
                    "method": "PUT",
                    "uri": f"{base}/productExtend/{existing_id}",
                    "data": payload,
                }
            )
        else:
            reqs.append(
                {
                    "method": "POST",
                    "uri": f"{base}/productExtend",
                    "data": payload,
                }
            )

    return reqs


def parse_batch_results(batch_response: Dict[str, Any]) -> List[Tuple[int, str]]:
    """
    EgyszerĹ± batch eredmĂ©ny parser.

    Kimenet:
        List[Tuple[int, str]]: [(statusCode, uri), ...]

    ElvĂˇrt input struktĂşra (gyakoribb Shoprenter vĂˇlasz):
        {
          "requests": {
            "request": [
              {
                "uri": "...",
                "response": {
                  "header": {"statusCode": 200},
                  ...
                }
              }
            ]
          }
        }

    ParamĂ©ter:
        batch_response (Dict[str, Any]): /batch vĂˇlasz JSON.

    VisszatĂ©rĂ©si Ă©rtĂ©k:
        List[Tuple[int, str]]: stĂˇtuszkĂłd + uri pĂˇrok.

    MegjegyzĂ©s:
        Ez a parser csak egy konkrĂ©t struktĂşrĂˇt kezel.
        Ha tĂ¶bbfĂ©le vĂˇlaszformĂˇtummal talĂˇlkozol, Ă©rdemes a korĂˇbbi
        "parse_batch_response" robusztusabb vĂˇltozatot hasznĂˇlni.
    """
    out: List[Tuple[int, str]] = []
    requests_block = batch_response.get("requests", {}).get("request", []) or []
    for item in requests_block:
        uri = item.get("uri", "")
        status = int(item.get("response", {}).get("header", {}).get("statusCode", 0) or 0)
        out.append((status, uri))
    return out

"""
shoprenter.client
=================

Shoprenter API kliens (minimĂˇl, de bĹ‘vĂ­thetĹ‘) HTTP Basic Auth-al.

Feladat
-------
Ez az osztĂˇly egysĂ©gesĂ­ti a Shoprenter API-hĂ­vĂˇsokat:
- session + auth + alap header beĂˇllĂ­tĂˇsok
- create / update mĹ±veletek a productExtend erĹ‘forrĂˇson
- SKU alapjĂˇn id keresĂ©s (ha a Shoprenter API tĂˇmogatja a szĹ±rĂ©st)
- lapozott lekĂ©rĂ©s a productExtend listĂˇbĂłl (sku_map Ă©pĂ­tĂ©shez)

MegjegyzĂ©s
----------
A Shoprenter API pontos szĹ±rĂ©si szintaxisa projektenkĂ©nt eltĂ©rhet.
A find_product_id_by_sku() implementĂˇciĂł ezĂ©rt "best effort" jellegĹ±:
- ha a /products endpoint nem tĂˇmogatja a sku paramot Ă©s 400-at ad,
  akkor None-t ad vissza.
"""

import requests
from requests.auth import HTTPBasicAuth
from typing import Dict, Any, Optional


class ShoprenterClient:
    """
    EgyszerĹ± Shoprenter API kliens.

    Konstruktor:
        ShoprenterClient(base_url, user, password)

    BelsĹ‘ mĹ±kĂ¶dĂ©s:
        - requests.Session-t hasznĂˇl ĂşjrahasznosĂ­thatĂł kapcsolatokhoz
        - HTTP Basic Auth-tal autentikĂˇl
        - JSON request/response header-eket ĂˇllĂ­t be

    AttribĂştumok:
        base_url (str):
            A Shoprenter API base URL, pl. "https://.../api".
            A vĂ©gĂ©rĹ‘l a "/" le van vĂˇgva.

        session (requests.Session):
            Az auth-olt session.

        timeout (int):
            Alap timeout mĂˇsodpercben (jelenleg 30).
    """

    def __init__(self, base_url: str, user: str, password: str):
        # Base URL normalizĂˇlĂˇsa: ne legyen a vĂ©gĂ©n "/"
        self.base_url = base_url.rstrip("/")

        # Session: connection pooling + kĂ¶zĂ¶s auth/header
        self.session = requests.Session()
        self.session.auth = HTTPBasicAuth(user, password)

        # Alap header-ek JSON API-hoz
        self.session.headers.update(
            {
                "Accept": "application/json",
                "Content-Type": "application/json",
            }
        )

        # Default timeout (hasznĂˇlva a get_product_extend_page-ben)
        self.timeout = 30

    def create_product(self, payload: Dict[str, Any]) -> Dict[str, Any]:
        """
        Ăšj termĂ©k lĂ©trehozĂˇsa (productExtend).

        ParamĂ©ter:
            payload (Dict[str, Any]):
                Shoprenter API kompatibilis create payload.

        VisszatĂ©rĂ©si Ă©rtĂ©k:
            Dict[str, Any]:
                A Shoprenter API JSON vĂˇlasza (jellemzĹ‘en tartalmaz "id"-t).

        KivĂ©tel:
            requests.HTTPError:
                ha a vĂˇlasz nem 2xx.
        """
        r = self.session.post(f"{self.base_url}/productExtend", json=payload)
        r.raise_for_status()
        return r.json()

    def find_product_id_by_sku(self, sku: str) -> Optional[str]:
        """
        TermĂ©k azonosĂ­tĂł (id) keresĂ©se SKU alapjĂˇn.

        FIGYELEM:
            A Shoprenter API szĹ±rĂ©si paramĂ©terei eltĂ©rhetnek.
            Itt egy gyakori mintĂˇt hasznĂˇlunk: GET /products?sku=...&limit=1&full=0

        ParamĂ©ter:
            sku (str): Keresett SKU.

        VisszatĂ©rĂ©si Ă©rtĂ©k:
            Optional[str]:
                - product id (string), ha talĂˇlat van
                - None, ha nincs talĂˇlat vagy a filter nem tĂˇmogatott

        MĹ±kĂ¶dĂ©s:
            - /products endpoint meghĂ­vĂˇsa
            - ha 400: feltĂ©telezzĂĽk, hogy a filter szintaxis nem okĂ© -> None
            - vĂˇlaszban:
                - items[0].href -> id kinyerĂ©se az URL vĂ©gĂ©bĹ‘l
                - vagy items[0].id

        KivĂ©tel:
            requests.HTTPError:
                egyĂ©b (nem 400) hibakĂłd esetĂ©n.
        """
        # A Shoprenter API ĂˇltalĂˇban tĂˇmogat szĹ±rĂ©st query parammal (sku=...)
        r = self.session.get(
            f"{self.base_url}/products",
            params={"limit": 1, "sku": sku, "full": 0},
        )

        if r.status_code == 400:
            # Ha nĂˇlatok mĂˇs filter szintaxis van, ezt kĂ©sĹ‘bb finomĂ­tjuk.
            return None

            return None

        href = items[0].get("href")
        if href:
            return href.rstrip("/").split("/")[-1]

        # vagy id mezĹ‘
        return items[0].get("id")

    def update_product(self, product_id: str, payload: Dict[str, Any]) -> Dict[str, Any]:
        """
        TermĂ©k frissĂ­tĂ©se (productExtend/{id}) PUT-tal.

        ParamĂ©terek:
            product_id (str):
                A frissĂ­tendĹ‘ productExtend azonosĂ­tĂłja.

            payload (Dict[str, Any]):

        VisszatĂ©rĂ©si Ă©rtĂ©k:
            Dict[str, Any]:
                Shoprenter API vĂˇlasz JSON-ja.

        KivĂ©tel:
            requests.HTTPError:
                ha a vĂˇlasz nem 2xx.
        """
        r = self.session.put(f"{self.base_url}/productExtend/{product_id}", json=payload)
        r.raise_for_status()
        return r.json()

    def get_product_extend_page(self, *, page: int, limit: int = 200, full: bool = True) -> dict:

        MegjegyzĂ©s:
            A timeout itt self.timeout Ă©rtĂ©ket hasznĂˇlja.
        """
        r = self.session.get(

            params={"page": page, "limit": limit, "full": 1 if full else 0},

        )
        r.raise_for_status()
        return r.json()

UPsert sorĂˇn el kell dĂ¶nteni:

from typing import Dict
import time


def build_product_sku_map(
    client,
    *,
    limit: int = 200,
    sleep_s: float = 0.2,
    max_pages: int = 2000,   # vĂ©szfĂ©k
) -> Dict[str, str]:
    """
    FelĂ©pĂ­ti a Shoprenter SKU -> productExtend id tĂ©rkĂ©pet.

    MĹ±kĂ¶dĂ©s:
        - lapozva meghĂ­vja a client.get_product_extend_page(...) metĂłdust
        - minden oldalon:
            - kiolvassa az items listĂˇt
            - minden elemnĂ©l:
                sku = item["sku"]
                id  = item["id"]
            - ha mindkettĹ‘ lĂ©tezik -> eltĂˇrolja a dict-ben
        - addig megy, amĂ­g:
            - el nem Ă©ri a pageCount-ot, vagy
            - el nem Ă©ri a max_pages limitet (vĂ©szfĂ©k)

    ParamĂ©terek:

            ShoprenterClient pĂ©ldĂˇny, amely biztosĂ­tja:
                get_product_extend_page(page, limit, full)

        limit (int):
            OldalmĂ©ret (hĂˇny termĂ©ket kĂ©r le egyszerre).
            Nagyobb Ă©rtĂ©k = kevesebb kĂ¶r, de nagyobb payload.

        sleep_s (float):
            VĂˇrakozĂˇs kĂ©t oldal lekĂ©rĂ©se kĂ¶zĂ¶tt (rate limit vĂ©delem).

        max_pages (int):
            BiztonsĂˇgi vĂ©szfĂ©k.
            Ha ennyi oldal utĂˇn mĂ©g nem Ă©rtĂĽnk a vĂ©gĂ©re,
            RuntimeError dobĂłdik (valĂłszĂ­nĹ±leg pageCount problĂ©ma).

    VisszatĂ©rĂ©si Ă©rtĂ©k:
        Dict[str, str]:
            SKU -> productExtend id map.

    PĂ©lda:
        {
          "ABC123": "4567",
          "XYZ999": "8910"
        }

    HibakezelĂ©s:
        - Ha page >= max_pages -> RuntimeError
        - A get_product_extend_page belĂĽl HTTPError dobhat, ha API hiba van.

    TeljesĂ­tmĂ©ny:
        - O(N) ahol N a shopban lĂ©vĹ‘ termĂ©kek szĂˇma.
        - 10-20k termĂ©k esetĂ©n Ă©rdemes a limit-et nĂ¶velni (pl. 500-1000),
          ha az API engedi.
    """
    page = 0
    out: Dict[str, str] = {}

    while True:
        t0 = time.time()

        # Oldal lekĂ©rĂ©se a Shoprenter API-bĂłl
        data = client.get_product_extend_page(
            page=page,
            limit=limit,
            full=True,   # full=True: biztosan legyen sku Ă©s id
        )

        # SKU -> id kinyerĂ©se
        for it in items:
            sku = (it.get("sku") or "").strip()
            pid = it.get("id")
            if sku and pid:
                out[sku] = pid

        dt = time.time() - t0

        print(
            f"[SKU_MAP] "
            f"page={page+1}/{page_count} "
            f"items={len(items)} "
            f"map_size={len(out)} "
            f"dt={dt:.2f}s"
        )

        page += 1

        # -----------------------------
        # STOP feltĂ©telek
        # -----------------------------

        # Ha az API jelzi a teljes oldalszĂˇmot, Ă©s elĂ©rtĂĽk a vĂ©gĂ©t
        if page_count and page >= page_count:
            break

        # VĂ©szfĂ©k: tĂşl sok oldal (valĂłszĂ­nĹ±leg pageCount/oldalazĂˇsi hiba)
        if page >= max_pages:
            raise RuntimeError(
                f"VĂ©szfĂ©k: max_pages elĂ©rve ({max_pages}). "
                "ValĂłszĂ­nĹ± pageCount/oldalazĂˇs gond van."

"""
shoprenter.upsert

EgyszerĹ± segĂ©dfĂĽggvĂ©nyek egyetlen termĂ©k CREATE / UPDATE (UPsert) mĹ±veletĂ©hez.

Feladat
-------
A SKU alapjĂˇn eldĂ¶nteni:
- ha mĂˇr lĂ©tezik a Shoprenterben (sku_map szerint) -> UPDATE
- ha nem lĂ©tezik -> CREATE

    SKU -> productExtend id

- ne kelljen minden SKU-ra kĂĽlĂ¶n lekĂ©rdezĂ©st indĂ­tani
- ugyanazon futĂˇson belĂĽl elkerĂĽljĂĽk a duplikĂˇlt create-et

----------
Nagy mennyisĂ©gĹ± termĂ©k esetĂ©n a batch alapĂş megoldĂˇs hatĂ©konyabb.
Ez a modul egyedi (soronkĂ©nti) hĂ­vĂˇsokhoz ideĂˇlis.

from typing import Dict, Any


def upsert_product(client, sku_map: dict[str, str], payload: Dict[str, Any]) -> str:
    """
    Egyetlen termĂ©k UPsert mĹ±velete SKU alapjĂˇn.

    MĹ±kĂ¶dĂ©s:
        1) SKU kiolvasĂˇsa a payload-bĂłl
        2) Ha SKU benne van a sku_map-ben:
              -> UPDATE (PUT /productExtend/{id})
        3) Ha nincs:
              -> CREATE (POST /productExtend)
              -> visszakapott id eltĂˇrolĂˇsa a sku_map-ben

    ParamĂ©terek:
        client:
            ShoprenterClient pĂ©ldĂˇny, amely biztosĂ­tja:
                - update_product(id, payload)
                - create_product(payload)

        sku_map (dict[str, str]):
            SKU -> productExtend id map.
            A create utĂˇn frissĂĽl, hogy a tovĂˇbbi mĹ±veleteknĂ©l
            ugyanazon futĂˇson belĂĽl mĂˇr UPDATE tĂ¶rtĂ©njen.

        payload (Dict[str, Any]):
            Shoprenter kompatibilis termĂ©k payload.
            KĂ¶telezĹ‘ kulcs: "sku"

    VisszatĂ©rĂ©si Ă©rtĂ©k:
        str:
            "updated" vagy "created"

    KivĂ©tel:
        KeyError:
            Ha a payload nem tartalmaz "sku" kulcsot.
        requests.HTTPError:
            Ha a Shoprenter API hibĂˇt ad.
    """
    sku = payload["sku"].strip()

    existing_id = sku_map.get(sku)

    if existing_id:
        # MĂˇr lĂ©tezĹ‘ termĂ©k -> UPDATE
        client.update_product(existing_id, payload)
        return "updated"

    # Ăšj termĂ©k -> CREATE
    created = client.create_product(payload)

    new_id = created.get("id")
    if new_id:
        # Map frissĂ­tĂ©se, hogy a futĂˇs tovĂˇbbi rĂ©szĂ©ben mĂˇr UPDATE tĂ¶rtĂ©njen
        sku_map[sku] = new_id

    return "created"


def upsert_product_with_map(client, sku_map: dict[str, str], payload: dict) -> str:
    """
    AlternatĂ­v, minimalista UPsert implementĂˇciĂł.

    FunkcionĂˇlisan nagyon hasonlĂł az upsert_product()-hez,
    de kevĂ©sbĂ© defensĂ­v (pl. nincs strip, nincs id None ellenĹ‘rzĂ©s).

    MĹ±kĂ¶dĂ©s:
        - Ha SKU benne van a sku_map-ben -> UPDATE
        - Ha nincs -> CREATE, majd az id eltĂˇrolĂˇsa a map-ben

        sku_map (dict[str, str]):

        str:
            "updated" vagy "created"

        ezĂ©rt production kĂ¶rnyezetben inkĂˇbb az upsert_product()
        ajĂˇnlott.

def reset_file(path: Path) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    path.write_text("", encoding="utf-8")


def append_jsonl(path: Path, obj: Any) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    with path.open("a", encoding="utf-8") as f:
        f.write(json.dumps(obj, ensure_ascii=False, indent=2))
        f.write("\n\n")


# -------------------------------------------------
# CSV
# -------------------------------------------------

def init_csv(path: Path, headers: list[str]) -> None:
    path.parent.mkdir(parents=True, exist_ok=True)
    with path.open("w", newline="", encoding="utf-8") as f:
        writer = csv.writer(f, delimiter=";")
        writer.writerow(headers)

    path.parent.mkdir(parents=True, exist_ok=True)
    with path.open("a", newline="", encoding="utf-8") as f:
        writer = csv.writer(f, delimiter=";")
        writer.writerow(row)

from __future__ import annotations

import re
import unicodedata


def norm_text(s: str) -> str:
    """Alap normalizĂˇlĂˇs: whitespace trim + belsĹ‘ space-ek Ă¶sszevonĂˇsa."""
    s = (s or "").strip()
    s = re.sub(r"\s+", " ", s)
    return s


def clean_sku(value: str) -> str:
    """
    SKU / model fĂˇjlnĂ©vhez:
    - trim
    - space -> -
    - csak biztonsĂˇgos karakterek: A-Z a-z 0-9 _ -
    """
    s = norm_text(value)
    if not s:
        return ""
    s = s.replace(" ", "-")
    s = re.sub(r"[^A-Za-z0-9_\-]", "", s)
    return s


def clean_folder_name(value: str) -> str:
    """
    MappanĂ©vhez (CSOPORT1):
    - Ă©kezetek eltĂˇvolĂ­tĂˇsa
    - szĂłkĂ¶z -> -
    - tiltott karakterek tĂ¶rlĂ©se
    """
    s = norm_text(value)
    if not s:
        return ""

    # Ă©kezetek le
    s = unicodedata.normalize("NFKD", s)
    s = "".join(ch for ch in s if not unicodedata.combining(ch))

    s = s.replace(" ", "-")
    s = re.sub(r"[^A-Za-z0-9_\-]", "", s)
    return s


def build_shop_image_path(csoport1: str, model: str, slot: int = 1, ext: str = ".jpg") -> str:
    folder = clean_folder_name(csoport1).lower()
    base = clean_sku(model)

    if not base or not folder:
        return ""

    if slot == 1:
        fname = f"{base}{ext}"
    else:
        fname = f"{base}-{slot-1}{ext}"

    return f"product/{folder}/{fname}"


def image_alt_from_model(name: str, model: str, *, keyword: str = "horgĂˇsz termĂ©k") -> str:
    """
    SEO-barĂˇt ALT:
    - tartalmazza a termĂ©knevet (ha van)
    - tartalmazza a modellt/cikkszĂˇmot (ha van)
    - opcionĂˇlis kulcsszĂł
    """
    name = norm_text(name)
    model = norm_text(model)

    parts = []
    if name:
        parts.append(name)
    if model:
        parts.append(model)

    base = " - ".join(parts).strip()

    if base and keyword:
        return f"{base} | {keyword}"
    if base:
        return base
    return "TermĂ©kkĂ©p"

from decimal import Decimal, ROUND_UP


VAT_RATE = Decimal("1.27")
ROUND_STEP = Decimal("5")


def net_to_gross_rounded_5(net: float | int | Decimal) -> float:
    """
    NettĂł -> bruttĂł (27% ĂFA) -> 5 Ft-ra felfelĂ© kerekĂ­tve.
    """
    net = Decimal(str(net))
    gross = net * VAT_RATE

    rounded = (
        (gross / ROUND_STEP)
        .to_integral_value(rounding=ROUND_UP)
        * ROUND_STEP
    )

    return float(rounded)

def create_client() -> ShoprenterClient:
    """

    """
    return ShoprenterClient(

# ==============================================================
"""

Kulcs:
    CLI-bĹ‘l hasznĂˇlhatĂł nĂ©v (--supplier)

Ă‰rtĂ©k:

SUPPLIERS = {

}

"""

Mindegyik action egy fĂĽggvĂ©ny,

def run_upsert(supplier, client):
    """Create + Update futtatĂˇsa."""
    supplier.run_upsert(client=client)


def run_delete(supplier, client):
    """Delete futtatĂˇsa."""
    supplier.run_delete(client=client)


def run_all(supplier, client):
    """Upsert majd Delete futtatĂˇsa."""
    supplier.run_upsert(client=client)
    supplier.run_delete(client=client)


ACTIONS: Dict[str, Callable] = {
    "upsert": run_upsert,
    "delete": run_delete,
    "all": run_all,
